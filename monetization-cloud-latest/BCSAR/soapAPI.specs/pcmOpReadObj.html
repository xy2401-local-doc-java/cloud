! pcmOpReadObj input parameter element specifications. Most of the fields on the
! return parameter element depend on the type of the object being read.
!
! Mandatory fields for input parameter element:
!	PIN_FLD_POID

! Poid of the object to be read.
field PIN_FLD_POID (
	type	=		PIN_FLDT_POID,
	perms	=		M,
);

! pcmOpReadObj return parameter element specifications. Details depend on the
! type of object being read.
!
! Mandatory fields for output parameter element:
!	PIN_FLD_POID
!	<all fields in the object read, details depend on object type>

! Poid of the object that was read.
field PIN_FLD_POID (
	type	=		PIN_FLDT_POID,
	perms	=		M,
);
